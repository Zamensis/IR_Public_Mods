yearly_country_pulse = {
	on_actions = {
		delay = { days = { 15 350 } }

	}

#-----------------------------------------------------------------------------------------

#-----------------------------------------------------------------------------------------
z_vanilla = {
	effect = {
		if = {
			limit = {
				owns = p:1
			}
			random = {
				chance = 5
				random_list = {
					75 = {
						random_country = {
							trigger_event = { id = z_vanilla.1 }
						}
					}
					25 = {
						trigger = {
							any_country = {
								is_ai = no
							}
						}
						random_country = {
							limit = {
								is_ai = no
							}
							trigger_event = { id = z_vanilla.1 }
						}
					}
				}
			}
		}
	}
}
