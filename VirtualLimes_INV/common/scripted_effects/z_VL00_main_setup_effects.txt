# --------------------------------------------------------------------------------
# Mark this game as using this mod, for compatibility purposes if needed
# --------------------------------------------------------------------------------
z_limes_setup_vflag_effect = {
	if = {
		limit = {
			NOT = { has_global_variable = z_limes_flag }
		}
		set_global_variable = {
			name = z_limes_flag
			value = z_limes_flag
		}
	}
}

# --------------------------------------------------------------------------------
# Set AI lucky factor for the rest of the game
# --------------------------------------------------------------------------------
z_limes_setup_lucky_effect = {
	if = {
		limit = {
			OR = {
				# Random values generated on game initialized will always be the same, so they must be re-generated after a new seed has been rolled (i.e. on the 1st year tick)
				current_date <= 451.1.1
				# For countries not present on game initialized
				NOT = { has_variable = z_limes_lucky_factor }
			}
		}
		set_variable = {
			name = z_limes_lucky_factor
			value = { 0 100 }
		}
	}
}

# --------------------------------------------------------------------------------
# Set starting truces
# --------------------------------------------------------------------------------

				}

			}

					target = c:PSS # Pissuria
					duration = 7300 # 20 years (capped at 8?)
				}
				add_truce = {
					target = c:ZNT # Zanthia
					duration = 7300 # 20 years (capped at 8?)
				}

					target = c:SCA # Sakia/Derbicia
					duration = 7300 # 20 years (capped at 8?)
				}
			}
