# --------------------------------------------------------------------------------
# Help the AI annex its many MMs
# --------------------------------------------------------------------------------
z_limes_annex_effect_default_b = {
	random_subject = {
		limit = {
			z_limes_be_annexed_trigger = yes
			any_owned_province = { z_limes_$CAPITAL$_trigger_$LEVEL$ = yes }
			OR = {
				has_coasts = yes
				any_neighbour_country = { this = root }
			}
		}
		every_owned_province = { set_owned_by = root }
	}
}
