#-----------------------------------------------------------------------------------------
# Vanilla on_actions
#-----------------------------------------------------------------------------------------
monthly_country_pulse = {
	on_actions = {
		z_airm_player # AI Royal Marriages for the player (monthly)
	}
}

yearly_country_pulse = {
	on_actions = {
		z_airm_ai # AI Royal Marriages for the AI (yearly)
	}
}

#-----------------------------------------------------------------------------------------
# AI Royal Marriages for the player (monthly)
#-----------------------------------------------------------------------------------------
z_airm_player = {
	effect = {
		if = {
			limit = {
				is_ai = no
				z_airm_country_can_receive_proposals_trigger = yes # Is this country eligible to receive RM proposals?
			}
		}
		z_airm_send_proposal_effect = yes
	}
}

#-----------------------------------------------------------------------------------------
# AI Royal Marriages for the AI (yearly)
#-----------------------------------------------------------------------------------------
z_airm_ai = {
	effect = {
		if = {
			limit = {
				is_ai = yes
				z_airm_country_can_receive_proposals_trigger = yes # Is this country eligible to receive RM proposals?
			}
		}
		z_airm_send_proposal_effect = yes
	}
}
