namespace = diadochi_flavor

diadochi_flavor.100 = {#NS
	type = country_event
	title = "diadochi_flavor.100.t"
	desc = "diadochi_flavor.100.desc"
	
	picture = steppe_cavalry
	
	left_portrait = root.current_ruler
	right_portrait = scope:origin.current_ruler
	
	trigger = {
		NOT = { has_global_variable = triggered_parthian_horde } # From Invictus
		OR = {
			has_variable = horde_invasion_origin
			AND = {
				country_culture_group = scythia
				is_ai = no
				num_of_cities >= 50
			}
		}
		war = no
		exists = c:SEL
		any_neighbour_country = {
			OR = {
				this = c:SEL
				AND = {
					exists = c:PAR
					this = c:PAR
				}
			}
		}
		c:SEL = {
			num_of_cities >= 60
			has_civil_war = no #Z
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		# From Invictus: "Added a medium sized army and give money to pay for it"
		set_global_variable = triggered_parthian_horde
		#hidden_effect = {
		#	add_country_modifier = {
		#		name = parthian_death_lasers
		#		duration = 3650
		#	}
		#}
        add_treasury = 750
        add_manpower = 20
        create_unit = {
            navy = no
            location = p:6657
            while = {
	            count = 6
	            add_subunit = light_infantry
            }
            while = {
                count = 15
                add_subunit = horse_archers
            }
            while = {
                count = 5
                add_subunit = light_cavalry
            }                                                    
            add_food = 100
            add_morale = 2
        }
        create_unit = {
            navy = no
            location = p:6812
            while = {
	            count = 6
	            add_subunit = light_infantry
            }
            while = {
                count = 10
                add_subunit = horse_archers
            }
            while = {
                count = 3
                add_subunit = light_cavalry
            }                                                    
            add_food = 100
            add_morale = 2
        }
		# End of Invictus addition
		c:SEL = {
			save_scope_as = origin
			current_ruler = {
				save_scope_as = sel_rul
			}
		}
		save_scope_as = target
		# From VL
		scope:origin = {
			ordered_owned_province = {
				limit = {
					OR = {
						is_in_region = persis_region
						is_in_region = media_region
					}
				}
				order_by = total_population
				save_scope_as = wargoal
			}
		}
		scope:wargoal.state = {
			every_state_province = { add_claim = root }
		}
		if = {
			limit = {
				root = { is_ai = yes }
				scope:origin = { is_ai = yes }
				z_limes_rules_buff_PAR = 1
			}
			region:media_region = {
				every_region_province = { add_claim = root }
			}
			scope:wargoal.region = {
				every_region_province = { add_claim = root }
			}
		}
		# End of from VL
	}
	
	option = {
		name = "diadochi_flavor.100.a"
		# From VL
		if = {
			limit = {
				root = { is_ai = yes }
				scope:origin = { is_ai = yes }
				z_limes_rules_buff_PAR = 1
				z_limes_rules_nerf_SEL = 1
			}
			declare_war_with_wargoal = {
				war_goal = conquer_wargoal
				target = scope:origin
				province = scope:wargoal
			}
		}
		else = {
			declare_war_with_wargoal = {
				war_goal = imperial_conquest_wargoal
				target = scope:origin
			}
		}
		# End of from VL
		add_manpower = 20 # from Invictus
		scope:origin = {
			#scope:sel_rul = {
			#	death = {
			#		death_reason = death_battle
			#		killer = scope:target.current_ruler
			#	}
			#}
			if = {
				limit = {
					is_ai = yes
				}
				add_stability = -15 # from Invictus
				add_stability = -10
				add_country_modifier = {
					name = parthian_attack_lasers
					duration = 3650
				}
				every_subject = { 
					limit = { is_ai = yes }
					add_stability = -10
					add_manpower = -10 # from Invictus
					add_country_modifier = {
						name = parthian_attack_lasers
						duration = 3650
					}
				}
			}
		}
		if = {
			limit = {
				scope:origin = {
					is_ai = yes
				}
				exists = c:PAR
				c:PAR = {
					is_ai = yes
					is_subject_of = scope:origin
				}
			}
			c:PAR = {
				every_owned_province = {
					set_owned_by = root
				}
			}
		}
		every_country = {
			limit = {
				NOT = {
					this = root
				}
			}
			trigger_event = {
				id = diadochi_flavor.101
			}
		}
		# From VL
		if = {
			limit = {
				root = { is_ai = yes }
				scope:origin = { is_ai = yes }
			}
			if = {
				limit = { z_limes_rules_buff_PAR = 1 }
				region:parthia_region = {
					every_region_province = {
						limit = { owned_or_subject_owned = scope:origin }
						set_owned_by = root
					}
				}
				area:rhagai_area = {
					every_area_province = {
						limit = { owned_or_subject_owned = scope:origin }
						set_owned_by = root
					}
				}
			}
			if = {
				limit = { z_limes_rules_nerf_SEL = 1 }
				scope:origin = {
					add_country_modifier = {
						name = z_limes_weak_ai
						duration = 1825 # 5 years
					}
				}
			}
		}
		# End of from VL
	}
	
	option = {
		name = "diadochi_flavor.100.b"
		
		current_ruler = {
			add_popularity = -25
		}
		
		ai_chance = {
			factor = 0
		}
		
	}
	# From Invictus
	after = {
		c:SEL = {
			trigger_event = diadochi_flavor.999
		}	
	}

}

# Invictus event

diadochi_flavor.999 = {#NS
	type = country_event
	hidden = yes

	trigger = {
		is_ai = yes
		c:BAC = {
			exists = yes
			is_ai = yes
		}
	}

	fire_only_once = yes
	
	immediate = {
		random_current_war = {
			limit = {
				any_war_participant = {
					this = c:SEL #Selucids
				}
				any_war_participant = {
					this = scope:target #Parthia
				}
				any_war_participant = {
					this = c:BAC #Bactria
				}
				is_war_leader = c:SEL
			}
			remove_from_war = c:BAC
		}
		c:BAC = {
			add_truce = {
				target = scope:target
				duration = 3650	#10 years
			}
		}
		add_political_influence = -25
	}
}
