# --------------------------------------------------------------------------------
# Base CCN effect
# --------------------------------------------------------------------------------
z_limes_CCN_effect = {
	if = {
		limit = {
			trigger_if = {
				limit = { is_ai = no }
				has_global_variable = z_limes_timer
				z_limes_year > 10
			}
			trigger_else = {
				z_limes_rules_CCN = 1
			}
			NOT = { has_variable = z_dynastic_countries_cooldown }
			NOT = { has_variable = z_limes_CCN_cooldown }
			is_republic = no
			is_subject = no
			has_civil_war = no
			z_limes_CCN_has_uneligible_culture_trigger = no
			z_limes_CCN_has_eligible_culture_trigger = yes
		}
		# Scenario 1: Give cultural name (first time, visible)
		if = {
			limit = {
				NOT = { has_variable = z_dynastic_countries_flag }
				NOT = { has_variable = z_limes_CCN_flag }
				# K name triggers only for monarchies
				trigger_if = {
					limit = { is_monarchy = yes }
					num_of_cities >= z_limes_defines_CCN_K
				}
				# E name triggers for any monarchy or tribe with >= 500 provinces
				trigger_else = {
					num_of_cities >= z_limes_defines_CCN_E
				}
			}
			trigger_event = { # Let the player/AI choose
				id = z_limes_CCN.1
				days = { 1 360 }
			}
		}
		# Scenario 2: Rename from K (hidden)
		else_if = {
			limit = {
				has_variable = z_limes_CCN_flag_k
				num_of_cities >= z_limes_defines_CCN_E
			}
			z_limes_CCN_rename_country_effect = yes
		}
	}
	# Remove harmless errors from log
	if = {
		limit = { always = no }
		set_variable = z_dynastic_countries_cooldown
		set_variable = z_dynastic_countries_flag
	}
}

#-----------------------------------------------------------------------------------------
# After the right family has been identified, actually change the country name.
#-----------------------------------------------------------------------------------------
z_limes_CCN_rename_this_country_effect = {
	if = {
		limit = {
			num_of_cities < z_limes_defines_CCN_E
		}
		change_country_name = $CULTURE$_CCN_K
		set_variable = z_limes_CCN_flag_k
		if = {
			limit = { has_variable = z_limes_CCN_flag_e }
			remove_variable = z_limes_CCN_flag_e
		}
	}
	else = {
		change_country_name = $CULTURE$_CCN_E
		set_variable = z_limes_CCN_flag_e
		if = {
			limit = { has_variable = z_limes_CCN_flag_k }
			remove_variable = z_limes_CCN_flag_k
		}
	}
	hidden_effect = { change_country_adjective = $CULTURE$_CCN_A }
}
